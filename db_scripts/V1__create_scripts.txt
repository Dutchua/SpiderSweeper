CREATE TABLE [User] (
  [UserID] int,
  [Username] varchar(50),
  PRIMARY KEY ([UserID])
);

CREATE TABLE [HighScore] (
  [HighScoreID] int,
  [UserID] int,
  [Score] int,
  [Date] datetime,
  PRIMARY KEY ([HighScoreID]),
  CONSTRAINT [FK_HighScore.UserID]
    FOREIGN KEY ([UserID])
      REFERENCES [User]([Username])
);
